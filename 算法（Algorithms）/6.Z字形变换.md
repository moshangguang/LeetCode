# Z字形变换

```
func convert(s string, numRows int) string {
	if len(s) < numRows {
		return s
	}
	if numRows == 1 {
		return s
	}
	zArray := make([][]int32, numRows)
	delta := 1
	index := 0
	data := make([]int32, 0, len(s))
	for _, c := range s {
		if zArray[index] == nil {
			zArray[index] = make([]int32, 0)
		}
		zArray[index] = append(zArray[index], c)
		if index == 0 {
			delta = 1
		}
		if index == numRows-1 {
			delta = -1
		}
		index += delta
	}
	for _, array := range zArray {
		for _, c := range array {
			data = append(data, c)
		}
	}
	return string(data)
}
```
